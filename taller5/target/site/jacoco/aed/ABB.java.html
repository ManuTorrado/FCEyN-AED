package aed;

import java.util.Stack;

public class ABB<T extends Comparable<T>> {
    private Nodo raiz;

    private class Nodo {
        private T value;
        private Nodo left;
        private Nodo right;

        public Nodo(T value) {
            this.value = value;
            this.left = null;
            this.right = null;
        }
    }

    public class HandleABB {
        private Nodo referencia;

        public HandleABB(Nodo referencia) {
            this.referencia = referencia;
        }

        public T valor() {
            return referencia.value;
        }

        public void eliminar() {
            ABB.this.eliminar(referencia.value);
        }

    }

    public ABB() {
        this.raiz = null;
    }

    public T maximo() {

        Nodo actual = this.raiz;
        while (actual.right != null) {
            actual = actual.right;
        }
        return actual.value;
    }

    public T minimo() {

        Nodo actual = this.raiz;
        while (actual.left != null) {
            actual = actual.left;
        }
        return actual.value;
    }

    public int cardinal() {
        return contarNodos(raiz);
    }

    private int contarNodos(Nodo n) {
        if (n == null) {
            return 0;
        }
        return 1 + contarNodos(n.left) + contarNodos(n.right);
    }

    public HandleABB insertar(T elem) {
        if (raiz == null) {
            raiz = new Nodo(elem);
            return new HandleABB(raiz);
        }

        Nodo actual = raiz;
        Nodo padre = null;

        while (actual != null) {
            padre = actual;
            if (elem.compareTo(actual.value) < 0)
                actual = actual.left;
            else
                actual = actual.right;
        }

        Nodo nuevo = new Nodo(elem);
        if (elem.compareTo(padre.value) < 0)
            padre.left = nuevo;
        else
            padre.right = nuevo;

        return new HandleABB(nuevo);
    }

    public boolean pertenece(T elem) {
        Nodo actual = raiz;
        while (actual != null) {
            int cmp = elem.compareTo(actual.value);
            if (cmp == 0)
                return true;
            actual = (cmp < 0) ? actual.left : actual.right;
        }
        return false;
    }

    /* eliminar, lo implemento con eliminacion recursiva */
    public void eliminar(T elem) {
        raiz = eliminarRec(raiz, elem);
    }

    private Nodo eliminarRec(Nodo nodo, T elem) {
        if (nodo == null)
            return null;

        int compartor = elem.compareTo(nodo.value);

        if (compartor < 0)
            nodo.left = eliminarRec(nodo.left, elem);
        else if (compartor > 0)
            nodo.right = eliminarRec(nodo.right, elem);
        else {
            // caso 1: sin hijos
            if (nodo.left == null && nodo.right == null)
                return null;

            // caso 2: un hijo
            if (nodo.left == null)
                return nodo.right;
            if (nodo.right == null)
                return nodo.left;

            // caso 3: dos hijos se reemplaza por el minimo derecho
            Nodo sucesor = nodo.right;
            while (sucesor.left != null)
                sucesor = sucesor.left;
            nodo.value = sucesor.value;
            nodo.right = eliminarRec(nodo.right, sucesor.value);
        }
        return nodo;
    }

    @Override
    public String toString() {
        String res = "{";
        ABB_Iterador it = iterador();

        if (it.haySiguiente()) {
            res = res + it.siguiente();
        }

        while (it.haySiguiente()) {
            res = res + ", " + it.siguiente();
        }

        res = res + "}";
        return res;
    }

    public class ABB_Iterador {
        private Stack<Nodo> pila = new Stack<>();

        public ABB_Iterador() {
            Nodo actual = raiz;
            while (actual != null) {
                pila.push(actual);
                actual = actual.left;
            }
        }

        public boolean haySiguiente() {
            return !pila.isEmpty();
        }

        public T siguiente() {
            Nodo n = pila.pop();
            T valor = n.value;

            Nodo actual = n.right;
            while (actual != null) {
                pila.push(actual);
                actual = actual.left;
            }

            return valor;
        }
    }

    public ABB_Iterador iterador() {
        return new ABB_Iterador();
    }

}
